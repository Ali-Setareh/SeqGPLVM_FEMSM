\begin{frame}{A simple scenario (blood pressure pill)}
Imagine:
\begin{itemize}
  \item $N$ individuals, $T=12$ months.
  \item Each month: take pill ($A_{it}=1$) or not ($A_{it}=0$).
  \item Measure outcome once at the end: $Y_{i,T+1}$ (blood pressure).
\end{itemize}

\vspace{0.2cm}
\textbf{RCT ideal:} randomize $A_{it}$ each month $\Rightarrow$
$\mathbb{E}\!\left[Y_{i,T+1}\mid\text{do}(\bar A_T=\bar a_T)\right]
=
\mathbb{E}\!\left[Y_{i,T+1}(\bar a_T)\right]
=
\mathbb{E}\!\left[Y_{i,T+1}\mid \bar A_T=\bar a_T\right]
$.


\vspace{0.2cm}
\textbf{Reality:} observational data $\Rightarrow$ treatment depends on health status, behavior, genetics, etc.
\end{frame}

% ==========================================================
\begin{frame}{What goes wrong: confounding}
Confounding = a third variable affects both treatment and outcome.

\vspace{0.2cm}
In our setting:
\begin{itemize}
  \item Measured confounders: observed covariates $X_{it}$ (e.g doctor visits).
  \item If we measure all the right covariates, we have sequential ignorability:
        $A_{it} \perp Y_{i,T+1}(\bar a_T) \mid \bar A_{i,t-1}, X_{it}$.
  \item We can use IPTW/MSMs to adjust for measured confounding and estimate causal effects: 
        $\mathbb{E}\!\left[Y_{i,T+1}(\bar a_T)\right] = \mathbb{E}\!\left[\frac{\mathbbm{1}\{\bar A_{iT}=\bar a_T\}}{\prod_t \Pr(A_{it}=a_{it} \mid \bar A_{i,t-1}, X_{it})} Y_{i,T+1}\right]$.
  \item \textbf{Intuition:} IPTW reweights the sample to create a pseudo-population where treatment is independent of confounders. 
  \item Positivity assumption: we need $0< \Pr(A_{it}=a_{it} \mid \bar A_{i,t-1}, X_{it}) < 1$ for all $a_{it}$ and covariate histories.
\end{itemize}
\end{frame}


\begin{frame}{Key challenge: time-invariant unmeasured confounding}
  
    
\textbf{Unmeasured time-invariant confounder:} $U_i$ (e.g., genetic predisposition / baseline severity).

\begin{frame}{Causal target}
Let $\bar A_{i,T}=(A_{i1},\dots,A_{iT})$ be treatment history, and $Y_{i,T+1}(\bar{A}_{T-k},\underbar{a}_{T-k})$ a \emph{truncated} potential outcome.


\vspace{0.2cm}
Example estimand (pill effect in last month, conditional on the past treatment history):
\[
\mathbb{E}\!\left[\,Y_{i,T+1}(\bar A_{T-1},1)-Y_{i,T+1}(\bar A_{T-1},0)\,\right].
\]

\vspace{0.2cm}
\textbf{Key identification condition:}\\
treatment is ``as-good-as-random'' given the past \emph{and} the missing time-invariant factor: 
\[A_{it} \perp Y_{i,T+1}(\bar a_T) \mid \bar A_{i,t-1}, X_{it}, U_i.\]

\textbf{problem:} $U_i$ is unobserved, so we can't directly adjust for it in the propensity model.

\end{frame}

% ==========================================================
\begin{frame}{Two methods, one shared pipeline}
Both approaches follow the same high-level recipe:

\vspace{0.2cm}
\textbf{Stage 1 (treatment model):}
estimate propensity scores
\[
\pi_{it} \approx \Pr(A_{it}=1 \mid \text{history}, X_{it}, \text{confounder substitute}).
\]

\vspace{0.2cm}
\textbf{Stage 2 (MSM + IPTW):}
construct weights $W_i=\prod_t w_{it}$ with $w_{it} = \frac{\mathbbm{1}\{A_{it} = a_{it}\}}{\pi_{it}^{a_{it}} (1-\pi_{it})^{1-a_{it}}}$ and fit a (truncated) marginal structural 
model for $Y_{i,T+1}(\bar a_{T-k}, \underbar{a}_{T-k})$. e.g.,
\[\mathbb{E}\!\left[Y_{i,T+1}(\bar a_{T-k}, \underbar{a}_{T-k})\right] = \alpha + \tau_F a_{iT} + \tau_C \sum_{t=T-k}^{T-1} a_{it}\]

\vspace{0.3cm}
\textbf{Difference:} how we build the ``confounder substitute'' used in Stage 1.
\end{frame}

% ==========================================================
\begin{frame}{Method 1: Fixed-effects MSM (IPTW-FE) \cite{blackwellEffectPoliticalAdvertising2024}}
\textbf{Idea:} put a unit-specific intercept in the propensity model to soak up all time-constant factors.

\vspace{0.2cm}
Typical example (logit):
\[
\text{logit}\,\Pr(A_{it}=1 \mid a_{i,t-1},x_{it},\alpha_i)
=
\alpha_i + \phi a_{i,t-1} + \beta^\top x_{it}.
\]

\vspace{0.2cm}
\textbf{Why it helps:} $\alpha_i$ can capture time-invariant measured + unmeasured confounding.

\vspace{0.2cm}
\textbf{Practical requirements:}
\begin{itemize}
  \item Enough within-unit treatment switching (not always treated / always control).
  \item convergence is relying on both $N\to\infty$ and $T\to\infty$ with fixed $N/T$ ratio.
  \item \textbf{Big limitation:} The propensity model is parametric and it needs to be correctly specified.
\end{itemize}
\end{frame}

% ==========================================================
\begin{frame}{Method 2: SeqGPLVM (Sequential Deconfounder)\cite{hattSequentialDeconfoundingCausal2024}}
\textbf{Idea:} infer a latent variable $Z_i$ from the \emph{entire treatment trajectory} (and covariates),
then condition on it in the treatment model.

\vspace{0.2cm}
Assumption: treatment assignment follows a \textbf{conditional Markov model} given $Z_i$:
\[
p(\bar a_{iT}\mid \bar x_{iT}, Z_i, \Theta)
=
p(a_{i1}\mid x_{i1}, Z_i,\theta_1)\prod_{t=2}^T p(a_{it}\mid a_{i,t-1},x_{it},Z_i,\theta_t).
\]

\vspace{0.2cm}
\textbf{Instantiation:} use Gaussian process latent variable models (GP priors for time-specific functions).
This makes the propensity model flexible/nonlinear: 
\[
p(\bar a_{iT}\mid \bar x_{iT}, z_i, \overline{\boldsymbol{f}})
=
p(a_{i1}\mid x_{i1}, z_i,f_1)\prod_{t=2}^T p(a_{it}\mid a_{i,t-1},x_{it},z_i,f_t).
\]
where $f_t \sim \mathcal{GP}(0, k_t(\cdot, \cdot))$. 
\end{frame}

\begin{frame}{Simulation study (aligned with Blackwell--Yamauchi)}
I benchmark both approaches on a known DGP with time-invariant unmeasured confounding:

\small
\begin{subequations}
\begin{align}
U_i &:= a(2V_i-1), \qquad V_i \sim \operatorname{Unif}(0,1)
\\
X_{it} &:= -\tfrac12\mathbbm{1} + L \xi_{it},
\qquad \xi_{it}\sim \mathcal{N}(0,I), \qquad LL^\top=\Sigma
\\
A_{it} &:= \mathbbm{1}\!\left\{\eta_{it} \le
\operatorname{expit}\!\left(U_i+\varphi A_{i,t-1}+\beta^\top X_{it}\right)\right\},
\qquad \eta_{it} \sim \operatorname{Unif}(0,1)
\\
Y_{iT} &:= U_i + \tau_F A_{iT} + \tau_C \sum_{t=T-3}^{T-1} A_{it}
+ \gamma^\top \bar X_{i} + \epsilon_{iT},
\qquad \epsilon_{iT} \sim \mathcal{N}(0,1).
\end{align}
\end{subequations}

\vspace{-0.25cm}
\footnotesize
\[
\operatorname{expit}(x)=\frac{1}{1+\exp(-x)}, \qquad
\bar X_{i}=\frac{1}{T}\sum_{t=1}^T X_{it}.
\]

\vspace{0.2cm}
Grid:
\begin{itemize}
  \item $N\in\{200,500,1000,3000\}$, and $\rho=N/T\in\{5,10,50\}$ (so $T$ varies with $N$).
  \item Low vs high heterogeneity via $a\in\{1,2\}$.
\end{itemize}
\end{frame}

\begin{frame}{Discussion}
\textbf{Fixed effects MSM:}
\begin{itemize}
  \item Needs switching; always-treated/control units cause in-sample positivity issues.
  \item Trimming / imputation is a bias--variance trade-off.
  \item It works only if the unmeasured confounder does not interact with time-varying covariates. 
\end{itemize}

\vspace{0.2cm}
\textbf{SeqGPLVM:}
\begin{itemize}
  \item More flexible treatment model; can capture nonlinearities and interactions. But since for each time we have a separate GP, this is the not the most efficient way for DGPs with smooth time dynamics.
  \item But can be sensitive to overfitting, leading to extreme weights and unstable estimates.
  \item Theory relies on assumptions like the conditional Markov structure.
  \item Strong assumption: Pinpointability. the latent variable must be identifiable from the treatment history. In another word, 
  the unmeasured confounder must be determinstic function of the treatment history (and covariates). This violates the positivity assumption. 
\end{itemize}
\end{frame}

% ==========================================================
\begin{frame}{Conclusion}
\textbf{Main message:} both methods try to reconstruct the missing time-invariant factor from treatment histories,
but they trade off simplicity vs flexibility.

\vspace{0.2cm}
\begin{itemize}
  \item IPTW-FE: simple, fast, interpretable; works well with enough $T$ and switching.
  \item SeqGPLVM: flexible treatment model; promising but can be sensitive (overfitting / weight instability / assumptions).
\end{itemize}

\vspace{0.3cm}
\textbf{Next steps:}
\begin{itemize}
  \item More diagnostics for weight stability + sensitivity to panel length.
  \item Clarify the mathematical assumptions between the two methods.
  \item Explore DGPs that stress-test key assumptions (e.g., pinpointability).
\end{itemize}
\end{frame}
